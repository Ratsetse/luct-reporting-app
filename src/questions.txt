export default function LecturerDashboard({ navigation }) {
  const { user } = useContext(AuthContext);
  const [courses, setCourses] = useState([]);

  // 📥 Fetch ONLY courses assigned to this lecturer
  const fetchCourses = async () => {
    const snapshot = await getDocs(collection(db, "courses"));

    const myCourses = snapshot.docs
      .map(doc => ({ id: doc.id, ...doc.data() }))
      .filter(course => course.lecturerId === user.uid);

    setCourses(myCourses);
  };

  useEffect(() => {
    fetchCourses();
  }, []);

  const handleLogout = async () => {
    await signOut(auth);
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>My Courses</Text>

      {courses.length === 0 ? (
        <Text>No courses assigned yet</Text>
      ) : (
        <FlatList
          data={courses}
          keyExtractor={(item) => item.id}
          renderItem={({ item }) => (
            <View style={styles.card}>
              <Text style={styles.course}>{item.name}</Text>

              <Button
                title="Create Report"
                onPress={() =>
                  navigation.navigate("CreateReport", {
                    courseId: item.id,
                    courseName: item.name,
                  })
                }
              />
            </View>
          )}
        />
      )}

      <Button title="Logout" onPress={handleLogout} color="red" />
    </View>
  );
}
